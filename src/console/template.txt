namespace gen
{

static const std::uint8_t gFileContents[FILE_SIZE] =
FILE_CONTENTS
/* size = FILE_SIZE bytes */

static const std::uint8_t gFileSha1[20U] = {FILE_SHA1};
static const std::uint32_t gFileCrc = FILE_CRC;

ArrayPtr<const std::uint8_t> GetFile()
{
    ArrayPtr<const std::uint8_t> file(gFileContents);
    return file;
}

ArrayPtr<const std::uint8_t> GetFileSignature()
{
	ArrayPtr<const std::uint8_t> signature(&gFileSha1[0], 20U);
	return signature;
}


std::uint32_t GetFileCrc()
{
    return gFileCrc;
}
